# List everything
ldapsearch -H ldap://127.0.0.1:1389 -b 'dc=example,dc=org' -x

# whoami/bind/authenticate
ldapwhoami -H ldap://127.0.0.1:1389 -D 'cn=user01,ou=users,dc=example,dc=org' -x -w bitnami1

# group search/list all groups a user is a member of
ldapsearch -H ldap://127.0.0.1:1389 -b "dc=example,dc=org" "(&(objectClass=groupOfNames)(member=cn=user01,ou=users,dc=example,dc=org))" -x

# modify user01
ldapmodify -H ldap://127.0.0.1:1389 -D "cn=admin,dc=example,dc=org" -w adminpassword -f user01.ldap
